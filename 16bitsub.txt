DATA SEGMENT 
	N1 DW 0003H
	N2 DW 0004H
	DIFF DW ?
	BORROW DB ?
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA

START:
	MOV AX, DATA
	MOV DS, AX
	
	MOV AX, N1
	SUB AX, N2
	
	MOV BL, 0
	JNC NOCARRY
	MOV BL, 1
	
	NOCARRY:
		MOV DIFF, AX
		MOV BORROW, BL
		
		MOV AH, 4CH
		INT 21H
CODE ENDS
END START
