DATA SEGMENT
    STRINGS DB 'CAT', 'DOG', 'BAT', 'RAT'
    LEN DB 4
    KEY DB 'FOX'
    MSG1 DB 10,13,'String Found$'
    MSG2 DB 10,13,'String Not Found$'
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA

START:
    MOV AX, DATA
    MOV DS, AX

    MOV CL, LEN
    LEA SI, STRINGS

OUTER_LOOP:
    MOV CH, 3
    LEA BX, KEY
    MOV DI, SI

COMPARE_LOOP:
    MOV AL, [DI]
    MOV DL, [BX]
    CMP AL, DL
    JNE NEXT_STRING
    INC DI
    INC BX
    DEC CH
    JNZ COMPARE_LOOP

    LEA DX, MSG1
    MOV AH, 09H
    INT 21H
    JMP EXIT

NEXT_STRING:
    ADD SI, 3
    DEC CL
    JNZ OUTER_LOOP

    LEA DX, MSG2
    MOV AH, 09H
    INT 21H

EXIT:
    MOV AH, 4CH
    INT 21H

CODE ENDS
END START
